# 10. Funciones javascript

* #### Obtener cuerpo en formato Json

```javascript
var jsonData = pm.response.json(); 
var user_id = jsonData.data.user_id;
```

* #### Obtener cuerpo en formato XML

```javascript
var responseText = pm.response.text(); 
var jsonData = xml2Json(responseText); 
var user_id = jsonData.data.user_id; 
```

* #### Agregar dato a una variable global

```javascript
pm.globals.set("user_id", "13376"); 
```

* #### Validar código de respuesta

```javascript
pm.test("Validar Código respuesta (200)", function () { 
pm.response.to.have.status(200); 
});
```

* #### Validar que las propiedades existan

```javascript
var jsonData = pm.response.json(); 
pm.test("Validar propiedades", function () {     
// Validar propiedades de primer nivel 
pm.expect(jsonData).to.have.property("code"); 
pm.expect(jsonData).to.have.property("rid"); 
pm.expect(jsonData).to.have.property("redirectUrl"); 
pm.expect(jsonData).to.have.property("data"); 
// Validar propiedades dentro de 'data' 
pm.expect(jsonData.data).to.have.property("auth_token"); 
pm.expect(jsonData.data).to.have.property("user_id"); 
pm.expect(jsonData.data).to.have.property("id_platform"); 
pm.expect(jsonData.data).to.have.property("channel_id"); 
pm.expect(jsonData.data).to.have.property("tokenSB"); 
});
```

* #### Si el cuerpo de la respuesta tiene un array

```javascript
pm.test("Primer usuario activo", function () { 
pm.expect(jsonData.users[0].active).to.be.true; 
});
```

* #### Validar si el cuerpo de la respuesta es un HTML

```javascript
pm.test("Validar que la respuesta sea HTML", function () { 
pm.response.to.have.header("Content-Type"); 
pm.expect(pm.response.headers.get("Content-Type")).to.include("text/html"); 
});
```

* #### Validar que el cuerpo de la respuesta no tenga un texto específico

```javascript
let html = pm.response.text(); 
pm.test("Validar que el juego esté disponible", function () { 
pm.expect(html).to.not.include("Juego No disponible"); 
}); 
```

* #### Validar tiempo de respuesta

```javascript
pm.test("Validar el tiempo de respuesta < 500ms", () => { 
pm.expect(pm.response.responseTime).to.be.below(500); 
});
```
